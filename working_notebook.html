# Zelle 11: Q9 — Final Equation
# Parameter des finalen Modells
final_model_params = lr_6.params

# Gleichung als String zusammenbauen
terms = []
for name, coef in final_model_params.items():
    if name == 'const':
        terms.append(f"{coef:.4f}")
    else:
        terms.append(f"{coef:.4f} * {name}")

final_equation = "price = " + " + ".join(terms)
